drop API


$('#draggable, #draggable-nonvalid').draggable()

$('#droppable').droppable({
	// 最基础的，当有东西被拖动到这里之后触发的函数
	drop: function(event, ui){
		$(this).addClass('text-danger').find('p').html('dropped!')
	},

	// 设置接受的元素, #id .class tag 都行
	accept: '#draggable',

	// 在可拖动元素被拖动的时候可以添加的class
	activeClass: 'bg-primary',

	// 拖动元素的中心点（x,y轴中心垂线交点）位于框内时可添加的class
	hoverClass: 'bg-danger'

})
